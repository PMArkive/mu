 3 # slow, iterative shift-left instruction
 4 # preconditions: _nr >= 0, _dr > 0
 5 fn repeated-shift-left nr: int, dr: int -> _/eax: int {
 6   var result/eax: int <- copy nr
 7   {
 8     compare dr, 0
 9     break-if-<=
10     result <- shift-left 1
11     decrement dr
12     loop
13   }
14   return result
15 }

31 fn abs n: int -> _/eax: int {
